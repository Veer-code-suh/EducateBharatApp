import React, { useEffect, useState } from 'react';
import {
    SafeAreaView,
    View,
    StyleSheet,
    Image,
    Text,
    Linking
} from 'react-native';
import {
    DrawerContentScrollView,
    DrawerItemList,
    DrawerItem
} from '@react-navigation/drawer';
import { SocialIcon, Icon } from 'react-native-elements';

import avatar from '../Assets/avatar.jpg';
import { COLOR } from '../Constants';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { useRecoilState } from 'recoil';
import { LoginState } from '../RecoilState/LoginState';
import { BACKEND_URL } from '@env';
import { useNavigation } from '@react-navigation/native';

const DrawerContent = (props) => {
    const navigation = useNavigation();
    const [userpic, setUserpic] = useState('');
    const [islogin, setIsLogin] = useRecoilState(LoginState);
    const [user, setUser] = useState({});

    const getUserDatafromToken = async () => {
        let token = await AsyncStorage.getItem("token");

        fetch(BACKEND_URL + "/getuserdatafromtoken", {
            method: "GET",
            headers: {
                "Content-Type": "application/json",
                "Authorization": `Bearer ${token}`
            }
        })
            .then(res => res.json())
            .then(data => {
                setUser(data.userdata);
                if (data.userdata.proilePic) {
                    setUserpic(data.userdata.proileImage);
                }
            });
    };

    useEffect(() => {
        getUserDatafromToken();
    }, [islogin]);

    const handleLogout = () => {
        AsyncStorage.removeItem("token");
        setIsLogin(false);
    };

    return (
        <SafeAreaView style={{ flex: 1 }}>
            {/* Profile Section */}
            <Image source={userpic ? { uri: userpic } : avatar} style={styles.sideMenuProfileIcon} />
            <Text style={styles.name}>{user?.name}</Text>
            <Text style={styles.email}>{user?.email}</Text>

            <View style={{ flex: 1 }}>
                <DrawerContentScrollView {...props}>
                    <DrawerItemList {...props} />
                    <DrawerItem labelStyle={styles.labelStyle}
                        style={styles.drawerItem} label="My Account" onPress={() => navigation.navigate("Settings")} />
                    <DrawerItem labelStyle={styles.labelStyle}
                        style={styles.drawerItem} label="My Courses" onPress={() => navigation.navigate("mycourses")} />
                    <DrawerItem labelStyle={styles.labelStyle}
                        style={styles.drawerItem} label="About Us" onPress={() => navigation.navigate("aboutus")} />
                </DrawerContentScrollView>
            </View>

            {/* Social Links Section */}
            <Text style={styles.contacttext}>Contact us</Text>
            <View style={styles.socialContainer}>
                <SocialIcon
                    type="instagram"
                    dark
                    raised={false}
                    iconSize={15}
                    onPress={() => Linking.openURL('https://www.instagram.com/educatebharatiitneet/')}
                />
                <SocialIcon
                    type="youtube"
                    raised={false}
                    iconSize={15}
                    onPress={() => Linking.openURL('https://www.youtube.com/@EducateBharatAmit')}
                />
                <SocialIcon
                    type="whatsapp"
                    raised={false}
                    iconSize={15}
                    onPress={() => Linking.openURL('https://wa.me/917897708104?text=Hi%20I%20am%20interested%20in%20your%20courses')}
                />
                <SocialIcon
                    type="linkedin"
                    raised={false}
                    iconSize={15}
                    onPress={() => Linking.openURL('https://www.linkedin.com/in/amit-maurya-8b0b4b221/')}
                />
                <Icon
                    name="phone"
                    type="font-awesome"
                    color="#f50"
                    raised
                    size={15}
                    onPress={() => Linking.openURL('tel:+917897708104')}
                />
            </View>

            {/* Logout Button */}
            <Text style={styles.logoutButton} onPress={handleLogout}>
                Logout
            </Text>

            <View style={{height:100}}></View>
        </SafeAreaView>
    );
};

const styles = StyleSheet.create({
    sideMenuProfileIcon: {
        width: 50,
        height: 50,
        borderRadius: 50,
        marginVertical: 20,
        alignSelf: "center"
    },
    name: {
        fontSize: 18,
        textAlign: "center",
        color: COLOR.col3
    },
    email: {
        fontSize: 14,
        textAlign: "center",
        color: COLOR.col2,
        marginBottom: 20
    },
    socialContainer: {
        flexDirection: 'row',
        flexWrap: 'wrap',
        justifyContent: "center",
        alignItems: "center",
        marginVertical: 5,
    },
    logoutButton: {
        fontSize: 16,
        fontWeight: '400',
        backgroundColor: COLOR.col6,
        color: 'white',
        borderRadius: 5,
        padding: 10,
        margin: 10,
        textAlign: "center"
    },
    drawerItem: {
        backgroundColor: '#FFF',
        borderRadius: 8,
        marginHorizontal: 10,
        marginVertical: 5,
    },
    labelStyle: {
        fontSize: 12,
        fontWeight:'300',
        color: COLOR.col2,
    },
    contacttext:{
        fontSize: 13,
        fontWeight:'400',
        color: '#10002B',
        alignSelf:'center'
    }
});

export default DrawerContent;
